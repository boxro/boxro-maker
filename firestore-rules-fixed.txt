      allow write: if request.auth != null && request.auth.uid == userId;

      allow update: if request.auth != null && 
        (request.auth.token.email in ["beagle3651@gmail.com", "boxro.crafts@gmail.com"] ||
         (request.resource.data.diff(resource.data).affectedKeys().hasOnly(['likes', 'shares', 'views', 'downloads', 'likedBy', 'sharedBy', 'downloadedBy', 'popularityBoost'])));
      allow delete: if request.auth != null && 
        (resource.data.authorId == request.auth.uid || 
         (resource.data.authorId == null && resource.data.authorEmail == request.auth.token.email) ||
         request.auth.token.email in ["beagle3651@gmail.com", "boxro.crafts@gmail.com"]);
    }
    
    // 박스로 톡 (갤러리 디자인에 대한 박스로 톡)
    match /boxroTalks/{document} {
      allow read: if true;

      allow update, delete: if request.auth != null && 
        (resource.data.authorId == request.auth.uid ||
         request.auth.token.email in ["beagle3651@gmail.com", "boxro.crafts@gmail.com"]);
    }
    
    // 박스로 톡 (스토리에 대한 박스로 톡)
    match /storyBoxroTalks/{document} {
      allow read: if true;
      allow create: if request.auth != null;
      allow update, delete: if request.auth != null && 
        (resource.data.authorId == request.auth.uid ||
         request.auth.token.email in ["beagle3651@gmail.com", "boxro.crafts@gmail.com"]);
    }
    
    // 박스로 톡 (스토어에 대한 박스로 톡)
    match /storeBoxroTalks/{document} {
      allow read: if true;
      allow create: if request.auth != null;
      allow update, delete: if request.auth != null && 
        (resource.data.authorId == request.auth.uid ||
         request.auth.token.email in ["beagle3651@gmail.com", "boxro.crafts@gmail.com"]);
    }
    
    // 박스카 이야기 글
    match /storyArticles/{document} {
      allow read: if true;
      allow create, delete: if request.auth != null && 
        request.auth.token.email in ["beagle3651@gmail.com", "boxro.crafts@gmail.com"];

        (request.auth.token.email in ["beagle3651@gmail.com", "boxro.crafts@gmail.com"] ||
         (request.resource.data.diff(resource.data).affectedKeys().hasOnly(['likes', 'shares', 'views', 'likedBy', 'sharedBy', 'viewedBy', 'popularityBoost'])));
    }
    
    // 스토어 아이템
    match /storeItems/{document} {
      allow read: if true;
      allow create, delete: if request.auth != null && 
        request.auth.token.email in ["beagle3651@gmail.com", "boxro.crafts@gmail.com"];
      allow update: if request.auth != null && 
        (request.auth.token.email in ["beagle3651@gmail.com", "boxro.crafts@gmail.com"] ||

    }
    
    // 박스카 이야기 댓글
    match /storyComments/{document} {
      allow read: if true;
      allow create: if request.auth != null;
      allow update, delete: if request.auth != null && resource.data.authorId == request.auth.uid;
    }
    
    // 사용자 상호작용 상태
    match /userInteractions/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;

    match /homeCards/{document} {
